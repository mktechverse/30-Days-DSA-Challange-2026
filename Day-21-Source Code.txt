public class MajorityElement {

    // Function to find the majority element
    public static int majorityElement(int[] nums) {

        // Step 1: Assume first element as candidate
        int candidate = nums[0];

        // Step 2: Initialize count as 1 (since we picked one element)
        int count = 1;

        // Step 3: Traverse the array from second element
        for (int i = 1; i < nums.length; i++) {

            // If current element is same as candidate
            if (nums[i] == candidate) {
                count++;  // Increase count (support increases)
            } 
            else {
                count--;  // Decrease count (opposition)
            }

            // If count becomes 0 → no clear winner so far
            if (count == 0) {
                // Pick new candidate
                candidate = nums[i];

                // Reset count
                count = 1;
            }
        }

        // Step 4: Return final candidate (majority element)
        return candidate;
    }

    public static void main(String[] args) {

        // Input array
        int[] nums = {2, 2, 1, 1, 1, 2, 2};

        // Call function
        int result = majorityElement(nums);

        // Print result
        System.out.println(result);
    }
}
