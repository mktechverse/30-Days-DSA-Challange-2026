public class TrappingRainWater {

    public static int trap(int[] height) {

        int left = 0;                      // left pointer (start of array)
        int right = height.length - 1;     // right pointer (end of array)

        int leftMax = 0;                   // highest wall seen so far from left
        int rightMax = 0;                  // highest wall seen so far from right

        int water = 0;                     // total water collected

        // loop until both pointers meet
        while (left < right) {

            // case 1: left wall is smaller
            if (height[left] < height[right]) {

                // if current height is greater than leftMax
                if (height[left] >= leftMax) {
                    leftMax = height[left];   // update leftMax
                } else {
                    // water trapped = leftMax - current height
                    water += leftMax - height[left];
                }

                left++;   // move left pointer forward

            } else {

                // case 2: right wall is smaller
                if (height[right] >= rightMax) {
                    rightMax = height[right]; // update rightMax
                } else {
                    // water trapped = rightMax - current height
                    water += rightMax - height[right];
                }

                right--;  // move right pointer backward
            }
        }

        return water;   // return total trapped water
    }

    public static void main(String[] args) {

        int[] height = {0,1,0,2,1,0,1,3,2,1,2,1};

        System.out.println(trap(height)); // expected output = 6
    }
}
